3 Parte A: Crear las Vistas del CMI (90 min)

3.1 Vista 1 — KPI Financiero: Ventas por Categoría

CREATE OR REPLACE VIEW vw_kpi_ventas_categoria AS
SELECT
    pr.categoria,
    YEAR(p.fecha_pedido)                            AS anio,
    MONTH(p.fecha_pedido)                           AS mes,
    SUM(oi.cantidad * oi.precio_unitario)           AS ventas_totales,
    COUNT(DISTINCT p.id_pedido)                     AS num_pedidos
FROM order_items oi
INNER JOIN pedidos   p  ON oi.id_pedido   = p.id_pedido
INNER JOIN productos pr ON oi.id_producto = pr.id_producto
WHERE p.estado = 'entregado'
GROUP BY pr.categoria, YEAR(p.fecha_pedido), MONTH(p.fecha_pedido)
ORDER BY anio DESC, mes DESC, ventas_totales DESC;

Verificar:

SELECT * FROM vw_kpi_ventas_categoria;

3.2 Vista 2 — KPI Cliente: Ticket Promedio por Región

CREATE OR REPLACE VIEW vw_kpi_ticket_region AS
SELECT
    c.region,
    COUNT(p.id_pedido)   AS total_pedidos,
    ROUND(AVG(p.total), 0) AS ticket_promedio,
    SUM(p.total)         AS ingresos_totales
FROM pedidos p
INNER JOIN clientes c ON p.id_cliente = c.id_cliente
WHERE p.estado = 'entregado'
GROUP BY c.region
ORDER BY ingresos_totales DESC;

3.3 Vista 3 — KPI Procesos: Tasa de Devolución por Zona

CREATE OR REPLACE VIEW vw_kpi_tasa_devolucion_zona AS
SELECT
    z.nombre                                   AS zona,
    COUNT(DISTINCT p.id_pedido)                AS total_pedidos,
    COUNT(DISTINCT d.id_devolucion)            AS total_devoluciones,
    ROUND(
        COUNT(DISTINCT d.id_devolucion) * 100.0 /
        NULLIF(COUNT(DISTINCT p.id_pedido), 0),
        2
    )                                          AS tasa_devolucion_pct
FROM zonas z
LEFT JOIN repartidores r ON z.id_zona    = r.id_zona
LEFT JOIN pedidos      p ON r.id_repartidor = p.id_repartidor
LEFT JOIN devoluciones d ON p.id_pedido  = d.id_pedido
GROUP BY z.id_zona, z.nombre
ORDER BY tasa_devolucion_pct DESC;

3.4 Vista 4 — KPI Aprendizaje: Clientes Inactivos

CREATE OR REPLACE VIEW vw_kpi_clientes_inactivos AS
SELECT
    c.id_cliente,
    c.nombre,
    c.email,
    c.region,
    MAX(p.fecha_pedido)                             AS ultimo_pedido,
    DATEDIFF(CURDATE(), MAX(p.fecha_pedido))        AS dias_sin_comprar
FROM clientes c
LEFT JOIN pedidos p ON c.id_cliente = p.id_cliente
GROUP BY c.id_cliente, c.nombre, c.email, c.region
HAVING dias_sin_comprar > 60
    OR MAX(p.fecha_pedido) IS NULL
ORDER BY dias_sin_comprar DESC;

3.5 Vista 5 — KPI Financiero: Producto Más Vendido por Mes

CREATE OR REPLACE VIEW vw_kpi_top_productos AS
SELECT
    pr.nombre                                    AS producto,
    pr.categoria,
    YEAR(p.fecha_pedido)                         AS anio,
    MONTH(p.fecha_pedido)                        AS mes,
    SUM(oi.cantidad)                             AS unidades_vendidas,
    SUM(oi.cantidad * oi.precio_unitario)        AS ingresos
FROM order_items oi
INNER JOIN productos pr ON oi.id_producto = pr.id_producto
INNER JOIN pedidos   p  ON oi.id_pedido   = p.id_pedido
WHERE p.estado = 'entregado'
GROUP BY pr.id_producto, pr.nombre, pr.categoria,
         YEAR(p.fecha_pedido), MONTH(p.fecha_pedido)
ORDER BY anio DESC, mes DESC, unidades_vendidas DESC;
